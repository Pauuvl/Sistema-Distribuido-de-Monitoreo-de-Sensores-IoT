/*
 * server.c — Servidor Central de Monitoreo IoT
 * Protocolo: IoTMP (IoT Monitoring Protocol)
 * Uso: ./server <puerto> <archivoDeLogs>
 *
 * Compilar: gcc -o server server.c -lpthread
 */

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <unistd.h>
#include <pthread.h>
#include <time.h>
#include <errno.h>
#include <sys/socket.h>
#include <sys/types.h>
#include <netinet/in.h>
#include <arpa/inet.h>

/* ─── Constantes ──────────────────────────────────────────────── */
#define MAX_CLIENTS     64
#define BUFFER_SIZE     512
#define TIMEOUT_SENSOR  60   /* segundos sin PING → inactivo */

/* ─── Umbrales de alerta (sección 7 del protocolo) ───────────── */
#define TEMP_HIGH       75.0
#define TEMP_LOW        -5.0
#define TEMP_CRITICAL   95.0
#define VIBR_HIGH       3.0
#define VIBR_CRITICAL   7.0
#define ENERGY_HIGH     400.0
#define ENERGY_CRITICAL 600.0

/* ─── Tipos de cliente ────────────────────────────────────────── */
#define TYPE_UNKNOWN    0
#define TYPE_SENSOR     1
#define TYPE_OPERATOR   2

/* ─── Estructura de un cliente conectado ─────────────────────── */
typedef struct {
    int         sockfd;
    int         active;         /* 1 = slot ocupado              */
    int         registered;     /* 1 = ya hizo REGISTER          */
    int         client_type;    /* TYPE_SENSOR / TYPE_OPERATOR   */
    char        id[64];         /* sensor-id u operator-id       */
    char        sensor_type[16];/* "temp", "vibr", "energy"      */
    double      last_value;     /* última medición recibida      */
    char        last_ts[32];    /* timestamp de última medición  */
    time_t      last_ping;      /* para detectar timeout         */
    char        ip[INET_ADDRSTRLEN];
    int         port;
} Client;

/* ─── Estado global ───────────────────────────────────────────── */
static Client          clients[MAX_CLIENTS];
static pthread_mutex_t clients_mutex = PTHREAD_MUTEX_INITIALIZER;
static FILE           *log_file      = NULL;
static pthread_mutex_t log_mutex     = PTHREAD_MUTEX_INITIALIZER;

/* ═══════════════════════════════════════════════════════════════
 *  LOGGING
 * ═══════════════════════════════════════════════════════════════ */
static void log_event(const char *ip, int port,
                      const char *received, const char *sent)
{
    time_t now = time(NULL);
    char   ts[32];
    strftime(ts, sizeof(ts), "%Y-%m-%dT%H:%M:%S", localtime(&now));

    pthread_mutex_lock(&log_mutex);

    /* Consola */
    printf("[%s] %s:%d | RX: %s | TX: %s\n", ts, ip, port, received, sent);
    fflush(stdout);

    /* Archivo */
    if (log_file) {
        fprintf(log_file, "[%s] %s:%d | RX: %s | TX: %s\n",
                ts, ip, port, received, sent);
        fflush(log_file);
    }

    pthread_mutex_unlock(&log_mutex);
}

static void log_info(const char *msg)
{
    time_t now = time(NULL);
    char   ts[32];
    strftime(ts, sizeof(ts), "%Y-%m-%dT%H:%M:%S", localtime(&now));

    pthread_mutex_lock(&log_mutex);
    printf("[%s] INFO: %s\n", ts, msg);
    fflush(stdout);
    if (log_file) {
        fprintf(log_file, "[%s] INFO: %s\n", ts, msg);
        fflush(log_file);
    }
    pthread_mutex_unlock(&log_mutex);
}

/* ═══════════════════════════════════════════════════════════════
 *  SOCKETS — helpers
 * ═══════════════════════════════════════════════════════════════ */

/* Lee una línea completa del socket hasta '\n' o EOF. */
static int read_line(int sockfd, char *buf, int maxlen)
{
    int  n = 0;
    char c;
    while (n < maxlen - 1) {
        int r = recv(sockfd, &c, 1, 0);
        if (r <= 0) return -1;   /* conexión cerrada o error */
        if (c == '\n') break;
        if (c == '\r') continue; /* ignorar \r si viniera */
        buf[n++] = c;
    }
    buf[n] = '\0';
    return n;
}

/* Envía una línea al socket (agrega '\n'). */
static int send_line(int sockfd, const char *msg)
{
    char buf[BUFFER_SIZE];
    int  len = snprintf(buf, sizeof(buf), "%s\n", msg);
    return send(sockfd, buf, len, 0);
}

/* ═══════════════════════════════════════════════════════════════
 *  GESTIÓN DE CLIENTES
 * ═══════════════════════════════════════════════════════════════ */

/* Reserva un slot libre; devuelve índice o -1 si está lleno. */
static int alloc_client(int sockfd, const char *ip, int port)
{
    pthread_mutex_lock(&clients_mutex);
    for (int i = 0; i < MAX_CLIENTS; i++) {
        if (!clients[i].active) {
            memset(&clients[i], 0, sizeof(Client));
            clients[i].active      = 1;
            clients[i].sockfd      = sockfd;
            clients[i].last_ping   = time(NULL);
            clients[i].port        = port;
            strncpy(clients[i].ip, ip, INET_ADDRSTRLEN - 1);
            pthread_mutex_unlock(&clients_mutex);
            return i;
        }
    }
    pthread_mutex_unlock(&clients_mutex);
    return -1;
}

/* Libera el slot de un cliente desconectado. */
static void free_client(int idx)
{
    pthread_mutex_lock(&clients_mutex);
    clients[idx].active     = 0;
    clients[idx].registered = 0;
    pthread_mutex_unlock(&clients_mutex);
}

/* Busca sensor por id; devuelve índice o -1. */
static int find_sensor(const char *sensor_id)
{
    for (int i = 0; i < MAX_CLIENTS; i++) {
        if (clients[i].active &&
            clients[i].client_type == TYPE_SENSOR &&
            strcmp(clients[i].id, sensor_id) == 0)
            return i;
    }
    return -1;
}

/* ═══════════════════════════════════════════════════════════════
 *  ALERTAS
 * ═══════════════════════════════════════════════════════════════ */

/* Determina el nivel de alerta según tipo y valor. */
static const char *check_alert(const char *tipo, double valor)
{
    if (strcmp(tipo, "temp") == 0) {
        if (valor > TEMP_CRITICAL) return "CRITICAL";
        if (valor > TEMP_HIGH)     return "HIGH";
        if (valor < TEMP_LOW)      return "LOW";
    } else if (strcmp(tipo, "vibr") == 0) {
        if (valor > VIBR_CRITICAL) return "CRITICAL";
        if (valor > VIBR_HIGH)     return "HIGH";
    } else if (strcmp(tipo, "energy") == 0) {
        if (valor > ENERGY_CRITICAL) return "CRITICAL";
        if (valor > ENERGY_HIGH)     return "HIGH";
    }
    return NULL;
}

/* Envía un mensaje ALERT a todos los operadores conectados. */
static void broadcast_alert(const char *sensor_id, const char *tipo,
                             const char *nivel, double valor,
                             const char *ts)
{
    char msg[BUFFER_SIZE];
    snprintf(msg, sizeof(msg),
             "ALERT %s %s %s %.2f %s",
             sensor_id, tipo, nivel, valor, ts);

    pthread_mutex_lock(&clients_mutex);
    for (int i = 0; i < MAX_CLIENTS; i++) {
        if (clients[i].active &&
            clients[i].registered &&
            clients[i].client_type == TYPE_OPERATOR) {
            send_line(clients[i].sockfd, msg);
            log_event(clients[i].ip, clients[i].port, "(broadcast)", msg);
        }
    }
    pthread_mutex_unlock(&clients_mutex);
}

/* ═══════════════════════════════════════════════════════════════
 *  PROCESAMIENTO DE COMANDOS
 * ═══════════════════════════════════════════════════════════════ */

/*
 * Procesa una línea recibida de un cliente.
 * Devuelve 0 para continuar, -1 para cerrar la conexión.
 */
static int process_message(int idx, const char *line)
{
    Client *c   = &clients[idx];
    char    response[BUFFER_SIZE];
    char    cmd[32];

    /* Parsear primer token (comando) */
    if (sscanf(line, "%31s", cmd) != 1) {
        send_line(c->sockfd, "ERROR INVALID_FORMAT");
        log_event(c->ip, c->port, line, "ERROR INVALID_FORMAT");
        return 0;
    }

    /* ── REGISTER ─────────────────────────────────────────────── */
    if (strcmp(cmd, "REGISTER") == 0) {
        char subtype[16], tipo[16], id[64];

        if (sscanf(line, "%*s %15s %15s %63s", subtype, tipo, id) == 3 &&
            strcmp(subtype, "SENSOR") == 0) {
            /* REGISTER SENSOR <tipo> <id> */
            if (strcmp(tipo, "temp")   != 0 &&
                strcmp(tipo, "vibr")   != 0 &&
                strcmp(tipo, "energy") != 0) {
                send_line(c->sockfd, "ERROR INVALID_FORMAT");
                log_event(c->ip, c->port, line, "ERROR INVALID_FORMAT");
                return 0;
            }
            c->registered   = 1;
            c->client_type  = TYPE_SENSOR;
            strncpy(c->id,          id,   sizeof(c->id)   - 1);
            strncpy(c->sensor_type, tipo, sizeof(c->sensor_type) - 1);
            c->last_ping = time(NULL);

            snprintf(response, sizeof(response), "OK REGISTERED %s", id);
            send_line(c->sockfd, response);
            log_event(c->ip, c->port, line, response);

        } else if (sscanf(line, "%*s %15s %63s", subtype, id) == 2 &&
                   strcmp(subtype, "OPERATOR") == 0) {
            /* REGISTER OPERATOR <id> */
            c->registered  = 1;
            c->client_type = TYPE_OPERATOR;
            strncpy(c->id, id, sizeof(c->id) - 1);

            snprintf(response, sizeof(response), "OK REGISTERED %s", id);
            send_line(c->sockfd, response);
            log_event(c->ip, c->port, line, response);

        } else {
            send_line(c->sockfd, "ERROR INVALID_FORMAT");
            log_event(c->ip, c->port, line, "ERROR INVALID_FORMAT");
        }
        return 0;
    }

    /* Cualquier otro comando requiere estar registrado */
    if (!c->registered) {
        send_line(c->sockfd, "ERROR AUTH_FAILED");
        log_event(c->ip, c->port, line, "ERROR AUTH_FAILED");
        return 0;
    }

    /* ── DATA ─────────────────────────────────────────────────── */
    if (strcmp(cmd, "DATA") == 0) {
        char   sensor_id[64], tipo[16], ts[32];
        double valor;

        if (sscanf(line, "%*s %63s %15s %lf %31s",
                   sensor_id, tipo, &valor, ts) != 4) {
            send_line(c->sockfd, "ERROR INVALID_FORMAT");
            log_event(c->ip, c->port, line, "ERROR INVALID_FORMAT");
            return 0;
        }

        /* Verificar que el sensor-id coincida con el registrado */
        if (strcmp(sensor_id, c->id) != 0) {
            send_line(c->sockfd, "ERROR SENSOR_NOT_FOUND");
            log_event(c->ip, c->port, line, "ERROR SENSOR_NOT_FOUND");
            return 0;
        }

        /* Actualizar última medición */
        pthread_mutex_lock(&clients_mutex);
        c->last_value = valor;
        strncpy(c->last_ts, ts, sizeof(c->last_ts) - 1);
        c->last_ping  = time(NULL);
        pthread_mutex_unlock(&clients_mutex);

        send_line(c->sockfd, "OK DATA_RECEIVED");
        log_event(c->ip, c->port, line, "OK DATA_RECEIVED");

        /* Verificar umbrales y alertar */
        const char *nivel = check_alert(tipo, valor);
        if (nivel) {
            broadcast_alert(sensor_id, tipo, nivel, valor, ts);
            char info[BUFFER_SIZE];
            snprintf(info, sizeof(info),
                     "ALERT generada: %s %s %s %.2f", sensor_id, tipo, nivel, valor);
            log_info(info);
        }
        return 0;
    }

    /* ── GET SENSORS ──────────────────────────────────────────── */
    if (strcmp(cmd, "GET") == 0) {
        char sub[16];
        if (sscanf(line, "%*s %15s", sub) != 1 ||
            strcmp(sub, "SENSORS") != 0) {
            send_line(c->sockfd, "ERROR UNKNOWN_COMMAND");
            log_event(c->ip, c->port, line, "ERROR UNKNOWN_COMMAND");
            return 0;
        }

        /* Construir lista de sensores activos */
        char   resp[BUFFER_SIZE];
        char   entry[128];
        int    found = 0;
        strcpy(resp, "SENSORS");

        pthread_mutex_lock(&clients_mutex);
        for (int i = 0; i < MAX_CLIENTS; i++) {
            if (clients[i].active &&
                clients[i].registered &&
                clients[i].client_type == TYPE_SENSOR) {
                snprintf(entry, sizeof(entry),
                         " %s:%s:%.2f",
                         clients[i].id,
                         clients[i].sensor_type,
                         clients[i].last_value);
                /* Comprobar que no sobrepasamos el buffer */
                if (strlen(resp) + strlen(entry) < BUFFER_SIZE - 2)
                    strcat(resp, entry);
                found = 1;
            }
        }
        pthread_mutex_unlock(&clients_mutex);

        if (!found) strcpy(resp, "SENSORS NONE");

        send_line(c->sockfd, resp);
        log_event(c->ip, c->port, line, resp);
        return 0;
    }

    /* ── PING ─────────────────────────────────────────────────── */
    if (strcmp(cmd, "PING") == 0) {
        pthread_mutex_lock(&clients_mutex);
        c->last_ping = time(NULL);
        pthread_mutex_unlock(&clients_mutex);

        send_line(c->sockfd, "PONG");
        log_event(c->ip, c->port, "PING", "PONG");
        return 0;
    }

    /* ── DISCONNECT ───────────────────────────────────────────── */
    if (strcmp(cmd, "DISCONNECT") == 0) {
        send_line(c->sockfd, "OK BYE");
        log_event(c->ip, c->port, "DISCONNECT", "OK BYE");
        return -1;   /* señal para cerrar el hilo */
    }

    /* ── Comando desconocido ──────────────────────────────────── */
    send_line(c->sockfd, "ERROR UNKNOWN_COMMAND");
    log_event(c->ip, c->port, line, "ERROR UNKNOWN_COMMAND");
    return 0;
}

/* ═══════════════════════════════════════════════════════════════
 *  HILO POR CLIENTE
 * ═══════════════════════════════════════════════════════════════ */
typedef struct { int idx; } ThreadArg;

static void *handle_client(void *arg)
{
    int idx = ((ThreadArg *)arg)->idx;
    free(arg);

    Client *c = &clients[idx];
    char    line[BUFFER_SIZE];

    {
        char info[128];
        snprintf(info, sizeof(info),
                 "Nueva conexión: %s:%d", c->ip, c->port);
        log_info(info);
    }

    while (1) {
        int n = read_line(c->sockfd, line, sizeof(line));
        if (n < 0) {
            /* Conexión cerrada por el cliente o error de red */
            char info[128];
            snprintf(info, sizeof(info),
                     "Desconexión abrupta: %s:%d", c->ip, c->port);
            log_info(info);
            break;
        }
        if (n == 0) continue;   /* línea vacía, ignorar */

        int ret = process_message(idx, line);
        if (ret < 0) break;     /* DISCONNECT recibido */
    }

    close(c->sockfd);
    free_client(idx);
    return NULL;
}

/* ═══════════════════════════════════════════════════════════════
 *  HILO DE WATCHDOG — detecta sensores sin PING
 * ═══════════════════════════════════════════════════════════════ */
static void *watchdog(void *arg)
{
    (void)arg;
    while (1) {
        sleep(10);   /* revisar cada 10 segundos */
        time_t now = time(NULL);

        pthread_mutex_lock(&clients_mutex);
        for (int i = 0; i < MAX_CLIENTS; i++) {
            if (clients[i].active &&
                clients[i].registered &&
                clients[i].client_type == TYPE_SENSOR) {
                double elapsed = difftime(now, clients[i].last_ping);
                if (elapsed > TIMEOUT_SENSOR) {
                    char info[128];
                    snprintf(info, sizeof(info),
                             "Sensor timeout: %s (%.0fs sin PING)",
                             clients[i].id, elapsed);
                    /* Loguear fuera del mutex para no bloquear, 
                       pero aquí lo hacemos simple */
                    printf("WATCHDOG: %s\n", info);
                    fflush(stdout);
                    if (log_file) {
                        fprintf(log_file, "WATCHDOG: %s\n", info);
                        fflush(log_file);
                    }
                    /* Marcar inactivo y cerrar socket */
                    close(clients[i].sockfd);
                    clients[i].active     = 0;
                    clients[i].registered = 0;
                }
            }
        }
        pthread_mutex_unlock(&clients_mutex);
    }
    return NULL;
}

/* ═══════════════════════════════════════════════════════════════
 *  MAIN
 * ═══════════════════════════════════════════════════════════════ */
int main(int argc, char *argv[])
{
    if (argc != 3) {
        fprintf(stderr, "Uso: %s <puerto> <archivoDeLogs>\n", argv[0]);
        return EXIT_FAILURE;
    }

    int         port     = atoi(argv[1]);
    const char *log_path = argv[2];

    if (port <= 0 || port > 65535) {
        fprintf(stderr, "Puerto inválido: %s\n", argv[1]);
        return EXIT_FAILURE;
    }

    /* Abrir archivo de logs */
    log_file = fopen(log_path, "a");
    if (!log_file) {
        perror("fopen log");
        return EXIT_FAILURE;
    }

    /* Inicializar array de clientes */
    memset(clients, 0, sizeof(clients));

    /* Crear socket del servidor */
    int server_fd = socket(AF_INET, SOCK_STREAM, 0);
    if (server_fd < 0) { perror("socket"); return EXIT_FAILURE; }

    /* Reusar dirección (evita "Address already in use" al reiniciar) */
    int opt = 1;
    setsockopt(server_fd, SOL_SOCKET, SO_REUSEADDR, &opt, sizeof(opt));

    struct sockaddr_in addr;
    memset(&addr, 0, sizeof(addr));
    addr.sin_family      = AF_INET;
    addr.sin_addr.s_addr = INADDR_ANY;
    addr.sin_port        = htons(port);

    if (bind(server_fd, (struct sockaddr *)&addr, sizeof(addr)) < 0) {
        perror("bind"); return EXIT_FAILURE;
    }

    if (listen(server_fd, 16) < 0) {
        perror("listen"); return EXIT_FAILURE;
    }

    {
        char info[64];
        snprintf(info, sizeof(info), "Servidor escuchando en puerto %d", port);
        log_info(info);
    }

    /* Lanzar hilo watchdog */
    pthread_t wt;
    pthread_create(&wt, NULL, watchdog, NULL);
    pthread_detach(wt);

    /* Bucle principal de aceptación */
    while (1) {
        struct sockaddr_in client_addr;
        socklen_t          client_len = sizeof(client_addr);

        int client_fd = accept(server_fd,
                               (struct sockaddr *)&client_addr,
                               &client_len);
        if (client_fd < 0) {
            perror("accept");
            continue;   /* no terminar el servidor por un error de accept */
        }

        char client_ip[INET_ADDRSTRLEN];
        inet_ntop(AF_INET, &client_addr.sin_addr, client_ip, sizeof(client_ip));
        int client_port = ntohs(client_addr.sin_port);

        int idx = alloc_client(client_fd, client_ip, client_port);
        if (idx < 0) {
            /* Servidor lleno */
            send_line(client_fd, "ERROR SERVER_FULL");
            close(client_fd);
            log_info("Conexión rechazada: servidor lleno");
            continue;
        }

        /* Crear hilo para este cliente */
        ThreadArg *targ = malloc(sizeof(ThreadArg));
        targ->idx = idx;

        pthread_t tid;
        if (pthread_create(&tid, NULL, handle_client, targ) != 0) {
            perror("pthread_create");
            free(targ);
            close(client_fd);
            free_client(idx);
            continue;
        }
        pthread_detach(tid);   /* el hilo se limpia solo al terminar */
    }

    fclose(log_file);
    close(server_fd);
    return EXIT_SUCCESS;
}
